195   /**
196    * @name $ADP.Util#loadScript
197    * @function
198    * @description Loads an external script & executes a callback function when an object, located in the 
199    *              external script & is returned in a <code>testFn</code>, is valid. 
200    * @param {string} objId Id used to identify script. Note: Implemented to avoid use of eval();
201    * @param {function} objReturnFn  Function that returns object to check against. Note: Implemented to avoid use of eval();
202    * @param {url} scriptSrc The url of the script to load.
203    * @param {string} callback The handler to be executed when script load is complete.
204    * @example
205    * $ADP.Util.loadScript('ExtScriptObj', function(){return ExtScriptObj;}, 'http://the.script.url/extobj.js', 
206    *   function(){
207    *      $ADP.Util.log('External script is done loading.');
208    *   }
209    * );
210    */  
211   _this.loadScript = function (objId, objReturnFn, scriptSrc, callback) {
212     var jsIntv;
213     var obj;
214 
215     /** If script is not currently being loaded, attempt to load. **/
216     var init = function() {
217       if (checkList(objId)) {
218         function wait() {
219           if(!checkList(objId)) {
220             clearInterval(waitIntv);
221             setScript();
222           }
223         }
224         var waitIntv = setInterval(wait, 100);
225       } else {
226         _loadList.push(objId);
227         setScript();
228       }
229     }
230 
231     /** Attempts to create script element if object does not exist. **/
232     var setScript = function() {
233       if(!checkObj()) {
234         var js = document.createElement('script');
235         js.setAttribute('type', 'text/javascript');
236         js.setAttribute('src', scriptSrc);
237         var s = document.getElementsByTagName('script')[0];
238         s.parentNode.insertBefore(js, s); 
239         jsIntv = setInterval(setObj, 100);
240       } else {
241         setObj();
242       } 
243     }
244 
245     /** Remove from check list executes callback. **/
246     var setObj = function() {
247       if(checkObj()) {
248         clearInterval(jsIntv);
249         removeFromList(objId);      
250         callback();
251       }
252     };
253 
254     /** Checks if function returns valid object. **/
255     var checkObj = function() { 
256       try { 
257           if(objReturnFn()) {
258           return true;
259         }
260        } catch (e) {
261         return false;
262        }
263     };
264 
265     /**
266      * Checks loadList for curreent IDs.
267      * @param {string} id String ID to check.
268      * @return Boolean
269      */
270      var checkList = function(id) {
271        for(var i=0; i < _loadList.length; i++) {
272          if (id == _loadList[i]) {
273            return true;
274          }
275        }
276        return false;
277      };
278      
279      /**
280      * Removes an ID from loadList.
281      * @param {string} id The string ID to remove.
282      */
283      var removeFromList = function(id) {
284        for(var i=0; i < _loadList.length; i++) {
285          if (id == _loadList[i]) {
286            _loadList = _loadList.slice(i, 1);
287            return;
288          }
289        }
290      };    
291     
292     init();
293   }

459   /**
460    * @name $ADP.PostMessageHandler#domRefPlayerWait
461    * @function
462    * @description Waits for DOM element to become available and then determines if iframe
463    *              verification is needed or message can be processed.
464    * @param {dom} dom DOM object this needs to be checked.
465    * @param {object} json JSON object to check or pass through.
466    */
467   _this.domRefPlayerWait = function (dom, json) {
468     _this.dom = dom;
469     _this.json = json;
470     
471     function iframeVerify(dom, json) {
472       var obj = new Object();
473       obj.postType = $ADP.PostMessage.INCOMING;
474       obj.uid = json.uid;
475       obj.fn = 'iframePlayerVerify';      
476       var player = _this.getPlayerByDomSearch(dom);
477       if (player) {
478         if (json.fn == "iframePlayerVerify") {
479           obj.params = true;
480           $ADP.PostMessage.send(obj, dom.contentWindow);   
481           return;
482         }
483         readyTest(dom, json, player);
484       } else {
485         obj.params = false;
486         $ADP.PostMessage.send(obj, dom.contentWindow);          
487       }
488     }
489     function iframeVerifyErr() {
490       $ADP.Util.log('Could not verify a parent iframe AdPlayer.');
491     }

496    * @name $ADP.PostMessageHandler#getPlayerByDomSearch
497    * @function
498    * @description Attempts to locate a parent AdPlayer from a DOM reference point.

500    */
501   _this.getPlayerByDomSearch = function (dom) {
502     var par = dom.parentNode;
503     while (!$ADP.AdPlayerManager.getAdPlayerById(par.id)) {
504       par = par.parentNode;
505       parName = par.nodeName.toLowerCase();
506       if ((parName == 'body') || (parName == 'html')) { break; }
507     }
508     if(par) {
509       var adPlayer = $ADP.AdPlayerManager.getAdPlayerById(par.id);
510       if(adPlayer) {
511         $ADP.Util.log('Found player at '+adPlayer.adDomElement().id);
512           return(adPlayer);
513       } else {
514         $ADP.Util.log('No AdPlayer found after parent search for "' + dom.id + '."');
515         return null;
516       }

520   /**
521    * @name $ADP.PostMessageHandler#readyTest
522    * @function
523    * @description Parses JSON object and executes requests.
524    * @param {dom} iframe Target iframe to communicate with.
525    * @param {object} json JSON object.
526    * @param {adplayer} player Current AdPlayer.
527    */
528   function readyTest(iframe, json, player) {
529     if (player) {
530       var params = json.params.split(',');
531       for (var t=0; t < params.length; t++) {
532         // Checks if it contains a function, which is needs to be properly wrapped and send off
533         switch (json.fn){
534           case 'addEventListener':
535             if (unescape(params[t]).match($ADP.PostMessage.FUNCTION)) {
536               var funcN = unescape(params[t]).slice($ADP.PostMessage.FUNCTION.length);
537               function funcMe (evt) {
538                 var obj = new Object();
539                 obj.postType = $ADP.PostMessage.INCOMING;
540                 obj.uid = json.uid;
541                 obj.fn = funcN;
542                 obj.evtType = evt.type();
543                 obj.uidName = json.uidName;
544                 $ADP.PostMessage.send(obj, iframe.contentWindow);
545               }
546               funcMe.uidName = json.uidName;
547               params[t] = funcMe;
548             }
549             break;
550         }
551       }
552       
553       switch (json.fn){
554         case 'removeEventListener':
555             params.push(json.uidName.toString());
556           break;
557         case 'track':
558           adEvtObj = new $ADP.AdEvent(params[0]);
559           adEvtObj.target(player);
560           adEvtObj.currentTarget(player);
561           player.track(new $ADP.AdEvent(params[0]), params[1]);
562           return;
563           break;
564       }
565 
566       player[json.fn].apply(player, params);
567     }
568   }

585         }

592       break;
593     default:

595       var player = $ADP.AdPlayerManager.getPlayerByUID(json.uid);  
596       // Need to target the current player and send back

598         var event = new $ADP.AdEvent(json.evtType);
599         event.target(player);
600         func(event);
601       }
602       break;
603     }
