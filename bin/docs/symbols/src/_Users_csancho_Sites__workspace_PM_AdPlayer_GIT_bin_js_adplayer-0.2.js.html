216     /**
217      * @private
218      * @description Dispatches all call-back function handlers added to the list.
219      */
220     function _dispatchCallBacks(adPlayer) {
221       var tmpLen = _callBackList.length;
222       var tempLenDiff = 0;
223       var index = 0;
224       do {
225         // call callback
226         if(typeof _callBackList[index] == 'function') {
227           _callBackList[index](adPlayer);
228         }
229         
230         // check if the temp length has changed
231         if(_callBackList > 0) {
232           tempLenDiff = tmpLen-_callBackList.length;
233           tmpLen = _callBackList.length;
234         } else {
235           tmpLen = 0;
236         }
237         
238         // if no difference then proceed to next index
239         if (tempLenDiff == 0) {
240           index++;
241         }
242       } while(index < tmpLen);
243     };

273 /**
274 * @class Responsible for handling the loading and referencing of a URL request.
275 * @description A pixel is requested by dynamically generating an <code>img</code>
276 *              element, which is appended to the document.  After appending is complete,
277 *              the <code>img</code> element is cleared from the document.
278 * 
279 * @author christopher.sancho@adtech.com
280 * 
281 * @param {string} url Optional - URL of the pixel to request.
282 * @property {string} url URL of the pixel to request.
283 * 
284 * @example
285 * var pixelRequest = new URLRequest('http://my.pixel-url.com');
286 * pixelRequest.load();
287 * 
288 * // Alternate
289 * var pixelRequest2 = new URLRequest();
290 * pixelRequest2.url = 'http://my.pixel-url.com';
291 * pixelRequest2.load();
292 */
293 function URLRequest(url) {
294   /** @property {string} The URL of the pixel to request. */
295   this.url;
296   if (url) { this.url = url; }
297 
298   /**
299    * @description Requests a pixel using the <code>url</code> property. 
300    */
301   this.load = function() {
302     if(this.url) {
303       var urlImgReq = document.createElement('img');
304       urlImgReq.src = (this.url);
305       urlImgReq.style.display = 'none';
306       var t = document.getElementsByTagName('script')[0];
307       log(this.url, 'URLRequest');
308       t.parentNode.insertBefore(urlImgReq, t);
309 
310       t.parentNode.removeChild(urlImgReq); // clear
311       delete urlImgReq;
312       delete t;
313     } else {
314       log('URLRequest', 'Parameter "url" is not defined.');
315     }
316   };

673  * @param callback {Function} The call-back handler to call when an <code>AdEvent</code> is dispatched. 
674  * 
675  * @see AdEvent
676  * @see AdPlayer#track
677  * 
678  * @example
679  * function trackEventHandler(adEvent) {
680  *  log(adEvent.type() + ' has been dispatched');
681  * }
682  * adPlayer.addAdEvent(AdEvent.TRACK, trackEventHandler);
683  */
684 AdPlayer.prototype.addAdEvent = function(adEvent, callback) {
685   if (!AdEvent.check(adEvent)) { return; }
686   if(!this.adEventListObj()[adEvent]) {
687     this.adEventListObj()[adEvent] = [];
688   }
689   // Check if the callback already exists, if not proceed.
690   for (var i = 0; i < this.adEventListObj()[adEvent].length; i++) {
691     if (this.adEventListObj()[adEvent][i] == callback ) {
692       return;      
693     }
694   }
695   this.adEventListObj()[adEvent].push(callback);
696 };

699 /** 
700  * @description Removes a callback function registered to an <code>AdEvent</code> flow.
701  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
702  * @param callback {Function} The call-back handler being called when an <code>AdEvent</code> is dispatched. 
703  * 
704  * @see AdEvent
705  * @see AdPlayer#track
706  * 
707  * @example
708  * function trackEventHandler(adEvent) {
709  *  // Remove call-back
710  *  adEvent.player().removeAdEvent(AdEvent.TRACK, trackEventHandler);
711  *  
712  *  log(adEvent.type() + ' has been dispatched');
713  * }
714  * adPlayer.addAdEvent(AdEvent.TRACK, trackEventHandler); 
715  */
716 AdPlayer.prototype.removeAdEvent = function(adEvent, callback) {
717   if (!AdEvent.check(adEvent)) { return; }
718   if (this.adEventListObj()[adEvent]) {
719     for (var i = 0; i < this.adEventListObj()[adEvent].length; i++) {
720       if (this.adEventListObj()[adEvent][i] == callback ) {
721         this.adEventListObj()[adEvent].splice(i, 1);
722         if (this.adEventListObj()[adEvent].length == 0) {
723           delete this.adEventListObj()[adEvent];
724         }
725         return;      
726       }
727     }
728   }
729 };

731 /** 
732  * @description A convenience function that adds a pixel URL to an <code>AdEvent</code> flow.  
733  *              
734  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
735  * @param url {String - URL} URL of pixel to call when associated <code>AdEvent</code> is dispatched.
736  * @param repeat {Boolean} Optional - Default is 'true.'  If set to 'false,' pixel will only fire once.
737  * 
738  * @see AdEvent
739  * @see AdPlayer#track
740  * @see URLRequest#load
741  * 
742  * @example
743  * // Adds a tracking pixel that will dispatch on AdEvent.TRACK event
744  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url');
745  * 
746  * // Adds a tracking pixel that will dispatch only once on AdEvent.TRACK event
747  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', null, false);
748  */
749 AdPlayer.prototype.addTrackingPixel = function(adEvent, url, repeat) {
750   if (!AdEvent.check(adEvent)) { return; }
751   if (repeat === undefined) { repeat = true; }
752   if (url) {
753     /** @private */
754     function defaultTrackCallBack(evt) {
755       var urlReq = new URLRequest(url);
756       urlReq.load();
757       if(!repeat) {
758     	evt.player().removeAdEvent(evt.type(), defaultTrackCallBack);
759       }
760     }
761     defaultTrackCallBack.url = url;
762     defaultTrackCallBack.repeat = repeat;
763     
764     this.addAdEvent(adEvent, defaultTrackCallBack);
765   } else {
766     log("Parameter 'url' must be defined", "addTrackingEvent");
767   }
768 };

770 /** 
771  * @description Removes a matching <code>url</code> or <code>callback</code> associated with
772  *              an <code>AdEvent</code> and registered using <code>addTrackingPixel()</code>
773  *              method.
774  *              <ul>
775  *                <li>If a <code>callback</code> and a <code>url</code> are both defined,
776  *                    method will remove the <code>AdEvent</code> associated with the callback.</li>
777  *                <li>If only a <code>callback</code> is defined method will remove the
778  *                    <code>AdEvent</code> associated with the callback.</li>
779  *                <li>If only a <code>url</code> is defined, method will remove all <code>AdEvent</code>
780  *                    callbacks containing a matching <code>url</code> property.</li>
781  *              </ul>
782  *              
783  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
784  * @param url {String - URL} Optional - URL of pixel to call when associated <code>AdEvent</code> is dispatched.
785  * 
786  * @see AdEvent
787  * @see AdPlayer#addTrackingPixel
788  * 
789  * @example
790  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url');
791  * adPlayer.addAdEvent(AdEvent.TRACK, removePixels);
792  * 
793  * function removePixels(adEvent) {
794  *  // Removes url from TRACK event flow.
795  *  adPlayer.removeTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url');
796  *  
797  * }
798  */
799 AdPlayer.prototype.removeTrackingPixel = function(adEvent, url) {
800   if (!AdEvent.check(adEvent)) { return; }
801   if (this.adEventListObj()[adEvent]) {
802     var tmpLen = this.adEventListObj()[adEvent].length;
803     var tempLenDiff = 0;
804     var index = 0;
805     do {
806       // Run through loop and remove all tracking that matches url
807       if (url) {
808         if (this.adEventListObj()[adEvent][index].name == 'defaultTrackCallBack') {
809           if (this.adEventListObj()[adEvent][index].url) { // remove all
810             if (this.adEventListObj()[adEvent][index].url == url) {
811               this.adEventListObj()[adEvent].splice(index, 1);
812               if (this.adEventListObj()[adEvent].length == 0) {
813                 delete this.adEventListObj()[adEvent];
814                 return;
815               }
816             }
817           }
818         }
819       }
820       
821       // check if the temp length has changed
822       if(this.adEventListObj()[adEvent]) {
823         tempLenDiff = tmpLen-this.adEventListObj()[adEvent].length;
824         tmpLen = this.adEventListObj()[adEvent].length;
825       } else {
826         tmpLen = 0;
827       }
828       
829       // if no difference then proceed to next index
830       if (tempLenDiff == 0) {
831         index++;
832       }
833     } while(index < tmpLen);
834   }
835 };

837 /** 
838  * @description Dispatches an <code>AdEvent</code> object to all suscribers.
839  *              
840  * @param adEventObj {AdEvent} The <code>AdEvent</code> instance to track.
841  * @param url {String - URL} Optional - URL of pixel that will be called once with track method.
842  * 
843  * @example
844  * // Dispatches a track event to all suscribers
845  * adPlayer.track(new AdEvent(AdEvent.TRACK));
846  * 
847  * // Dispatches a track event to all suscribers and calls URL once.
848  * adPlayer.track(new AdEvent(AdEvent.TRACK), 'http://my.pixel.url');
849  */
850 AdPlayer.prototype.track = function(adEventObj, url) {
851   if (!AdEvent.check(adEventObj.type())) { return; }
852   log(adEventObj.type(), 'track');
853   if (this.adEventListObj()[adEventObj.type()]) {
854     var tmpLen = this.adEventListObj()[adEventObj.type()].length;
855     var tempLenDiff = 0;
856     var index = 0;
857     do {
858       // call callback
859       adEventObj.player(this);
860       if (url) {
861         adEventObj.data().url = url;
862       }
863       this.adEventListObj()[adEventObj.type()][index](adEventObj);
864       
865       // check if the temp length has changed
866       if(this.adEventListObj()[adEventObj.type()]) {
867         tempLenDiff = tmpLen-this.adEventListObj()[adEventObj.type()].length;
868         tmpLen = this.adEventListObj()[adEventObj.type()].length;
869       } else {
870         tmpLen = 0;
871       }
872       
873       // if no difference then proceed to next index
874       if (tempLenDiff == 0) {
875         index++;
876       }
877     } while(index < tmpLen);
878   }
879   if (url) {
880     var urlReq = new URLRequest(url);
881     urlReq.load();
882   }
883 };

945 /**
946  * @description Convenience function that creates a layer that displays the privacy info added to <code>privacyInfoList</code>. <br/>
947  *              <code>AdEvent.PRIVACY_CLOSE</code> is dispatched when method is called. <br/>
948  *              Layer should be styled to preference using css.
949  *              <ul>
950  *                <li>.privacyPanel</li>
951  *                <li>.privacyPanel ul</li>
952  *                <li>.privacyPanel ul li</li>
953  *                <li>.privacyPanel button</li>
954  *              </ul>
955  * 
956  * @see AdPlayer#privacyInfoList
957  * @see PrivacyInfo
958 */
959 AdPlayer.prototype.showPrivacyInfo = function() {
960   if (!this.privacyPanel) {
961     this.privacyPanel = document.createElement('div');
962     this.privacyPanel.setAttribute('class', 'privacyPanel');
963     var privacyPanelList = document.createElement('ul');
964     var privacyPanelClose = document.createElement('button');
965     var parentThis = this;
966     this.privacyPanel.appendChild(privacyPanelList);
967     this.privacyPanel.appendChild(privacyPanelClose);
968     this.adDomElement().appendChild(this.privacyPanel);
969     privacyPanelClose.innerHTML = 'Close';
970     for (var i = 0; i < this.privacyInfoList().length; i++) {
971       var privacyElement =  document.createElement('li');
972       privacyClick = function(url) {
973         var data = new Object();
974         data.url = url;
975         parentThis.track(new AdEvent(AdEvent.PRIVACY_CLICK, data));
976         window.open(url);
977       }
978       privacyElement.innerHTML = this.privacyInfoList()[i].adServer + '<span>'+this.privacyInfoList()[i].message+'<br/><a href="javascript:privacyClick(\''+this.privacyInfoList()[i].url+'\');" target="_self">Opt Out</a></span>';
979       privacyPanelList.appendChild(privacyElement);
980     }
981     this.isPrivacyPanelEnabled(true);
982     privacyPanelClose.onclick = function() {
983       parentThis.hidePrivacyInfo();      
984     };
985     this.track(new AdEvent(AdEvent.PRIVACY_OPEN));
986   }
987 };
