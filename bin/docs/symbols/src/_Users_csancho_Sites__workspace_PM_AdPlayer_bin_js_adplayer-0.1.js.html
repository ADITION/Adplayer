174     /**
175      * @private
176      * @description Dispatches all call-back function handlers added to the list.
177      */
178     function _dispatchCallBacks(adPlayer) {
179       var tmpLen = _callBackList.length;
180       var tempLenDiff = 0;
181       var index = 0;
182       do {
183         // call callback
184         if(typeof _callBackList[index] == 'function') {
185           _callBackList[index](adPlayer);
186         }
187         
188         // check if the temp length has changed
189         if(_callBackList > 0) {
190           tempLenDiff = tmpLen-_callBackList.length;
191           tmpLen = _callBackList.length;
192         } else {
193           tmpLen = 0;
194         }
195         
196         // if no difference then proceed to next index
197         if (tempLenDiff == 0) {
198           index++;
199         }
200       } while(index < tmpLen);
201     };

231 /**
232 * @class Responsible for handling the loading and referencing of a URL request.
233 * @description A pixel is requested by dynamically generating an <code>img</code>
234 *              element, which is appended to the document.  After appending is complete,
235 *              the <code>img</code> element is cleared from the document.
236 * 
237 * @author christopher.sancho@adtech.com
238 * 
239 * @param {string} url Optional - URL of the pixel to request.
240 * @property {string} url URL of the pixel to request.
241 * 
242 * @example
243 * var pixelRequest = new URLRequest('http://my.pixel-url.com');
244 * pixelRequest.load();
245 * 
246 * // Alternate
247 * var pixelRequest2 = new URLRequest();
248 * pixelRequest2.url = 'http://my.pixel-url.com';
249 * pixelRequest2.load();
250 */
251 function URLRequest(url) {
252   /** @property {string} The URL of the pixel to request. */
253   this.url;
254   if (url) { this.url = url; }
255 
256   /**
257    * @description Requests a pixel using the <code>url</code> property. 
258    */
259   this.load = function() {
260     if(this.url) {
261       var urlImgReq = document.createElement('img');
262       urlImgReq.src = (this.url);
263       urlImgReq.style.display = 'none';
264       var t = document.getElementsByTagName('script')[0];
265       log(this.url, 'URLRequest');
266       t.parentNode.insertBefore(urlImgReq, t);
267 
268       t.parentNode.removeChild(urlImgReq); // clear
269       delete urlImgReq;
270       delete t;
271     } else {
272       log('URLRequest', 'Parameter "url" is not defined.');
273     }
274   };
275 }

554 /** 
555  * @description Adds a callback function to an <code>AdEvent</code> flow. Call-back handler function 
556  *              must expect a parameter that accepts an <code>AdPlayer</code> instance.
557  *              
558  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
559  * @param callback {Function} The call-back handler to call when an <code>AdEvent</code> is dispatched. 
560  * 
561  * @see AdEvent
562  * @see AdPlayer#track
563  * 
564  * @example
565  * function trackEventHandler(adPlayer) {
566  *  log('AdEvent.TRACK' has been dispatched');
567  * }
568  * adPlayer.addAdEvent(AdEvent.TRACK, trackEventHandler);
569  */
570 AdPlayer.prototype.addAdEvent = function(adEvent, callback) {
571   if (!AdEvent.check(adEvent)) { return; }
572   if(!this.adEventListObj()[adEvent]) {
573     this.adEventListObj()[adEvent] = [];
574   }
575   // Check if the callback already exists, if not proceed.
576   for (var i = 0; i < this.adEventListObj()[adEvent].length; i++) {
577     if (this.adEventListObj()[adEvent][i] == callback ) {
578       return;      
579     }
580   }
581   this.adEventListObj()[adEvent].push(callback);
582 };

585 /** 
586  * @description Removes a callback function registered to an <code>AdEvent</code> flow.
587  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
588  * @param callback {Function} The call-back handler being called when an <code>AdEvent</code> is dispatched. 
589  * 
590  * @see AdEvent
591  * @see AdPlayer#track
592  * 
593  * @example
594  * function trackEventHandler(adPlayer) {
595  *  // Remove call-back
596  *  adPlayer.removeAdEvent(AdEvent.TRACK, trackEventHandler);
597  *  
598  *  log('AdEvent.TRACK' has been dispatched');
599  * }
600  * adPlayer.addAdEvent(AdEvent.TRACK, trackEventHandler); 
601  */
602 AdPlayer.prototype.removeAdEvent = function(adEvent, callback) {
603   if (!AdEvent.check(adEvent)) { return; }
604   if (this.adEventListObj()[adEvent]) {
605     for (var i = 0; i < this.adEventListObj()[adEvent].length; i++) {
606       if (this.adEventListObj()[adEvent][i] == callback ) {
607         this.adEventListObj()[adEvent].splice(i, 1);
608         if (this.adEventListObj()[adEvent].length == 0) {
609           delete this.adEventListObj()[adEvent];
610         }
611         return;      
612       }
613     }
614   }

631  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url');
632  * 
633  * // Adds a tracking pixel that will dispatch only once on AdEvent.TRACK event
634  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', null, false);
635  *

645  * }
646  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', trackEventHandler, false);
647  */
648 AdPlayer.prototype.addTrackingPixel = function(adEvent, url, callback, repeat) {
649   if (!AdEvent.check(adEvent)) { return; }
650   if (repeat === undefined) { repeat = true; }
651   
652   if(callback) {
653     this.addAdEvent(adEvent, callback);
654   }
655   if (url) {
656     /** @private */
657     defaultTrackCallBack = function(adPlayer) {
658       if(adPlayer) {
659         var trackingUrl = url;
660         var urlReq = new URLRequest(trackingUrl);
661         urlReq.load();
662         if(!repeat) {
663           adPlayer.removeAdEvent(adEvent, defaultTrackCallBack);
664         }
665       }
666       return url;
667     }
668     this.addAdEvent(adEvent, defaultTrackCallBack);
669   } else {
670     log("Parameter 'url' must be defined", "addTrackingEvent");
671   }
672 };

674 /** 
675  * @description Removes a matching <code>url</code> or <code>callback</code> associated with
676  *              an <code>AdEvent</code> and registered using <code>addTrackingPixel()</code>
677  *              method.
678  *              <ul>
679  *                <li>If a <code>callback</code> and a <code>url</code> are both defined,
680  *                    method will remove the <code>AdEvent</code> associated with the callback.</li>
681  *                <li>If only a <code>callback</code> is defined method will remove the
682  *                    <code>AdEvent</code> associated with the callback.</li>
683  *                <li>If only a <code>url</code> is defined, method will remove all <code>AdEvent</code>
684  *                    callbacks containing a matching <code>url</code> property.</li>
685  *              </ul>
686  *              
687  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
688  * @param url {String - URL} Optional - URL of pixel to call when associated <code>AdEvent</code> is dispatched.
689  * @param callback {Function} Optional - The call-back handler to call when an <code>AdEvent</code> is dispatched.
690  * 
691  * @see AdEvent
692  * @see AdPlayer#addTrackingPixel
693  * 
694  * @example
695  * // Adds a tracking pixel, with a call-back, and will dispatch on AdEvent.TRACK event
696  * function trackEventHandler(adPlayer) {
697  *  // Removes pixel tracking associated with callback
698  *  adPlayer.removeTrackingPixel(AdEvent.TRACK, null, trackEventHandler);
699  *  
700  *  log('AdEvent.TRACK' has been dispatched');
701  * }
702  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', trackEventHandler);
703  *
704  * 
705  * // Remove all callbacks associated with pixel url
706  * function trackEventHandler1(adPlayer) {
707  *  log('AdEvent.TRACK' has been dispatched');
708  * }
709  * function trackEventHandler2(adPlayer) {
710  *  // Removes pixel tracking associated with callback
711  *  log('AdEvent.TRACK' has been dispatched');
712  * }
713  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', trackEventHandler1);
714  * adPlayer.addTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url', trackEventHandler2);
715  * 
716  * function removePixels() {
717  *  // Removes 'trackEventHandler1' and 'trackEventHandler2' from AdEvent.TRACK event flow
718  *  adPlayer.removeTrackingPixel(AdEvent.TRACK, 'http://my.pixel.url');
719  * }
720  * 
721  * removePixels();
722  */
723 AdPlayer.prototype.removeTrackingPixel = function(adEvent, url, callback) {
724   if (!AdEvent.check(adEvent)) { return; }
725   if (this.adEventListObj()[adEvent]) {
726     var tmpLen = this.adEventListObj()[adEvent].length;
727     var tempLenDiff = 0;
728     var index = 0;
729     do {
730       // Remove if callback matches and stop loop
731       if (callback) {
732         if (this.adEventListObj()[adEvent][index] == callback ) {
733           this.adEventListObj()[adEvent].splice(index, 1);
734           if (this.adEventListObj()[adEvent].length == 0) {
735             delete this.adEventListObj()[adEvent];
736           }
737           return; 
738         }
739       }  
740       // Run through loop and remove all tracking that matches url
741       if (url) {
742         if (this.adEventListObj()[adEvent][index](null)) { // remove all
743           if (this.adEventListObj()[adEvent][index](null) == url) {
744             this.adEventListObj()[adEvent].splice(index, 1);
745             if (this.adEventListObj()[adEvent].length == 0) {
746               delete this.adEventListObj()[adEvent];
747               return;
748             }
749           }
750         }
751       }
752       
753       // check if the temp length has changed
754       if(this.adEventListObj()[adEvent]) {
755         tempLenDiff = tmpLen-this.adEventListObj()[adEvent].length;
756         tmpLen = this.adEventListObj()[adEvent].length;
757       } else {
758         tmpLen = 0;
759       }
760       
761       // if no difference then proceed to next index
762       if (tempLenDiff == 0) {
763         index++;
764       }
765     } while(index < tmpLen);
766   }
767 };

769 /** 
770  * @description Dispatches an <code>AdEvent</code> string to all suscribers.
771  *              
772  * @param adEvent {AdEvent} The <code>AdEvent</code> to listen to.
773  * @param url {String - URL} Optional - URL of pixel that will be called once with track method.
774  * 
775  * @example
776  * // Dispatches a track event to all suscribers
777  * adPlayer.track(AdEvent.TRACK);
778  * 
779  * // Dispatches a track event to all suscribers and calls URL once.
780  * adPlayer.track(AdEvent.TRACK, 'http://my.pixel.url');
781  */
782 AdPlayer.prototype.track = function(adEvent, url) {
783   if (!AdEvent.check(adEvent)) { return; }
784   log(adEvent, 'track');
785   if (this.adEventListObj()[adEvent]) {
786     var tmpLen = this.adEventListObj()[adEvent].length;
787     var tempLenDiff = 0;
788     var index = 0;
789     do {
790       // call callback
791       this.adEventListObj()[adEvent][index](this);
792       
793       // check if the temp length has changed
794       if(this.adEventListObj()[adEvent]) {
795         tempLenDiff = tmpLen-this.adEventListObj()[adEvent].length;
796         tmpLen = this.adEventListObj()[adEvent].length;
797       } else {
798         tmpLen = 0;
799       }
800       
801       // if no difference then proceed to next index
802       if (tempLenDiff == 0) {
803         index++;
804       }
805     } while(index < tmpLen);
806   }
807   if (url) {
808     var urlReq = new URLRequest(url);
809     urlReq.load();
810   }
811 };

873 /**
874  * @description Convenience function that creates a layer that displays the privacy info added to <code>privacyInfoList</code>. <br/>
875  *              <code>AdEvent.PRIVACY_CLOSE</code> is dispatched when method is called. <br/>
876  *              Layer should be styled to preference using css.
877  *              <ul>
878  *                <li>.privacyPanel</li>
879  *                <li>.privacyPanel ul</li>
880  *                <li>.privacyPanel ul li</li>
881  *                <li>.privacyPanel button</li>
882  *              </ul>
883  * 
884  * @see AdPlayer#privacyInfoList
885  * @see PrivacyInfo
886 */
887 AdPlayer.prototype.showPrivacyInfo = function() {
888   if (!this.privacyPanel) {
889     this.privacyPanel = document.createElement('div');
890     this.privacyPanel.setAttribute('class', 'privacyPanel');
891     var privacyPanelList = document.createElement('ul');
892     var privacyPanelClose = document.createElement('button');
893     var parentThis = this;
894     this.privacyPanel.appendChild(privacyPanelList);
895     this.privacyPanel.appendChild(privacyPanelClose);
896     this.adDomElement().appendChild(this.privacyPanel);
897     privacyPanelClose.innerHTML = 'Close';
898     for (var i = 0; i < this.privacyInfoList().length; i++) {
899       var privacyElement =  document.createElement('li');
900       privacyClick = function(url) {
901         parentThis.track(AdEvent.PRIVACY_CLICK);
902         window.open(url);
903       }
904       privacyElement.innerHTML = this.privacyInfoList()[i].adServer + '<span>'+this.privacyInfoList()[i].message+'<br/><a href="javascript:privacyClick(\''+this.privacyInfoList()[i].url+'\');" target="_self">Opt Out</a></span>';
905       privacyPanelList.appendChild(privacyElement);
906     }
907     this.isPrivacyPanelEnabled(true);
908     privacyPanelClose.onclick = function() {
909       parentThis.hidePrivacyInfo();      
910     };
911     this.track(AdEvent.PRIVACY_OPEN);
912   }
913 };

915 /**
916  * @description Used in conjunction with <code>showPrivacyInfo</code> to remove privacy info layer.<br/>
917  *              <code>AdEvent.PRIVACY_CLOSE</code> is dispatched when method is called.
918  * 
919  * @see AdPlayer#showPrivacyInfo
920  * @see PrivacyInfo

924     this.isPrivacyPanelEnabled(false);

926     this.privacyPanel = null;
927     this.track(AdEvent.PRIVACY_CLOSE);
928   }
